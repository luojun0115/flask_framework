if __name__ == '__main__':
    # db.create_all()

    # 创建示例数据
    bk1 = Book(title='老王回忆录')
    bk2 = Book(title='我读书少，你别骗我')
    bk3 = Book(title='如何才能让自己更骚')
    bk4 = Book(title='怎样征服美丽少女')
    bk5 = Book(title='如何征服英俊少男')
    # 把数据提交
    db.session.add_all([bk1, bk2, bk3, bk4, bk5])
    db.session.commit()

    app.run()
