  • 点击添加按钮时，完成添加处理，修改视图add，完善post逻辑的代码，代码如下：

@app.route('/add', methods=['GET', 'POST'])
def add():
    if request.method == 'GET':
        return render_template('d1_add.html')
    elif request.method == 'POST':
        # 接收表单提交的数据
        title = request.form.get('title')
        # 创建对象
        book = Book()
        book.title = title
        # 提交到数据库
        db.session.add(book)
        db.session.commit()
        # 返回首页
        return redirect(url_for('index'))
